[docs]    def start(self):
        """
        start the watchdog
        """
        if self.timeout > 0:
            self.timer = Timer(self.timeout, self.handler)



[docs]    def stop(self):
        """
        stop the watchdog
        """
        if self.timer is not None:
            self.timer.cancel()
            self.timer = None
